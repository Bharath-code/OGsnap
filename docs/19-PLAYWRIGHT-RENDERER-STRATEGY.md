## 4. Implementation Details

### Complete Renderer Service

```typescript
// renderer/src/index.ts
import { chromium, Browser, Page } from 'playwright';
import { uploadToR2 } from './storage';

interface RenderOptions {
  html: string;
  width?: number;
  height?: number;
  options?: {
    addWatermark?: boolean;
    watermarkText?: string;
    quality?: 'low' | 'medium' | 'high';
  };
}

class Renderer {
  private browser: Browser | null = null;
  private isWarming = false;

  async init() {
    if (this.browser) return;
    
    this.browser = await chromium.launch({
      headless: true,
      args: [
        '--no-sandbox',
        '--disable-setuid-sandbox',
        '--disable-dev-shm-usage',
        '--disable-accelerated-2d-canvas',
        '--no-first-run',
        '--no-zygote',
        '--single-process',
        '--disable-gpu',
      ],
    });
  }

  async warmup() {
    if (this.isWarming) return;
    this.isWarming = true;

    try {
      // Create a warm page
      const page = await this.browser!.newPage();
      await page.setContent('<html><body>Warmup</body></html>');
      await page.close();
    } finally {
      this.isWarming = false;
    }
  }

  async render(options: RenderOptions): Promise<string> {
    await this.init();

    const {
      html,
      width = 1200,
      height = 630,
      options: renderOptions = {},
    } = options;

    const page = await this.browser!.newPage({
      viewport: { width, height },
    });

    try {
      // Set content
      await page.setContent(html, {
        waitUntil: 'networkidle',
      });

      // Wait for fonts/images to load
      await page.waitForTimeout(500);

      // Wait for specific elements if needed
      await page.waitForSelector('.og-content', { timeout: 2000 }).catch(() => {
        // Continue even if element not found
      });

      // Take screenshot
      const screenshot = await page.screenshot({
        type: 'png',
        fullPage: false,
      });

      // Add watermark if needed
      const finalImage = renderOptions.addWatermark
        ? await this.addWatermark(screenshot, renderOptions.watermarkText)
        : screenshot;

      // Upload to R2
      const imageUrl = await uploadToR2(finalImage);

      return imageUrl;
    } finally {
      await page.close();
    }
  }

  private async addWatermark(imageBuffer: Buffer, text = 'ogsnap.dev'): Promise<Buffer> {
    // Use sharp to add watermark
    const sharp = require('sharp');
    
    const watermarkSvg = `
      <svg width="1200" height="630">
        <text 
          x="1180" 
          y="620" 
          font-family="Inter, sans-serif" 
          font-size="14" 
          fill="white" 
          fill-opacity="0.5"
          text-anchor="end"
        >${text}</text>
      </svg>
    `;

    return await sharp(imageBuffer)
      .composite([{
        input: Buffer.from(watermarkSvg),
        top: 0,
        left: 0,
      }])
      .toBuffer();
  }

  async close() {
    if (this.browser) {
      await this.browser.close();
      this.browser = null;
    }
  }
}

export const renderer = new Renderer();
```

### HTTP Handler

```typescript
// renderer/src/handler.ts
import { VercelRequest, VercelResponse } from '@vercel/node';
import { render } from './renderer';

export default async function handler(
  req: VercelRequest,
  res: VercelResponse
) {
  if (req.method !== 'POST') {
    return res.status(405).json({ error: 'Method not allowed' });
  }

  try {
    const { html, width, height, options } = req.body;

    if (!html) {
      return res.status(400).json({ error: 'HTML is required' });
    }

    // Render with timeout
    const imageUrl = await Promise.race([
      render({ html, width, height, options }),
      new Promise((_, reject) =>
        setTimeout(() => reject(new Error('Render timeout')), 10000)
      ),
    ]);

    return res.status(200).json({ url: imageUrl });
  } catch (error) {
    console.error('Render error:', error);
    return res.status(500).json({ error: 'Render failed' });
  }
}
```
